{{'STEP_1' | translate}}


var xhttp = new XMLHttpRequest();
xhttp.onreadystatechange = function() {
    if (xhttp.readyState == 4 &&
                xhttp.status == 200) {
        var songs =
                JSON.parse(xhttp.responseText);
        showSongs(songs);
    }
};
xhttp.open("GET", "songs.json", true);
xhttp.send();

{{'STEP_2' | translate}}


var showSongs = function(songs)
{
    var div = document.getElementById('songs');
    var innerHtml = '
    ';
    for (var i = 0; i < songs.length; i++)
    {
        innerHtml += '
  • ' + songs[i].author +
                ': ' + songs[i].title + '
    • ';
    }
    innerHtml += '
';
    div.innerHTML = innerHtml;
}

var app = angular.module('ng-presentation', []);

app.controller('MainController',
                function($scope, $html)
{
    $html.get('songs.json').then(function(res)
    {
        $scope.songs = res.data;
    });
});
