var xhttp = new XMLHttpRequest();
xhttp.onreadystatechange = function() {
    if (xhttp.readyState == 4 &&
            xhttp.status == 200) {
        var songs =
            JSON.parse(xhttp.responseText);
        showSongs(songs);
    }
};
xhttp.open("GET", "endpoint/index.php", true);
xhttp.send();

{{'STEP_2' | translate}}


var showSongs = function(songs)
{
  var div = document.getElementById('songs');
  var innerHtml = '
    ';
  for (var i = 0; i < songs.length; i++)
  {
      innerHtml += '
  • ' + songs[i].author +
              ': ' + songs[i].title + '
    • ';
  }
  innerHtml += '
';
  div.innerHTML = innerHtml;
}

{{'CREATE_SONG' | translate}}


// Inside the controller
$scope.new_song = {
	author: '',
	title: ''
};
            

$scope.addSong = function()
{
    var author = $scope.new_song.author;
    var title = $scope.new_song.title;
    $http.post('endpoint/index.php',
        {author: author, title: title}).then(
    function(res)
    {
        $scope.songs.push({
            author: author,
            title: title,
            id: res.data
        });
        $scope.new_song.author = '';
        $scope.new_song.title = '';
    });
}

{{'SERVICES' | translate}}


getAll: function()
{
    var deferred = $q.defer();
    $http.get(ENDPOINT).then(function(res)
    {
        deferred.resolve(res.data);
    }, function()
    {
        deferred.reject();
    });
    return deferred.promise;
}
